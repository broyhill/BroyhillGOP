## SQL: AUTO-COUNT EACH SEGMENT IN YOUR SPINE

```sql
-- Count actual spine members per segment (run against your live data)
SELECT
    'Farmers' as segment,
    COUNT(DISTINCT ps.person_id) as in_spine,
    COUNT(DISTINCT CASE WHEN ps.total_contributed > 0 THEN ps.person_id END) as donors,
    ROUND(SUM(ps.total_contributed)) as total_raised
FROM core.person_spine ps
WHERE ps.employer_sic BETWEEN '0100' AND '0999'
   OR ps.employer_norm ILIKE '%farm%'
   OR ps.employer_norm ILIKE '%agriculture%'

UNION ALL SELECT 'Doctors',
    COUNT(DISTINCT ps.person_id),
    COUNT(DISTINCT CASE WHEN ps.total_contributed > 0 THEN ps.person_id END),
    ROUND(SUM(ps.total_contributed))
FROM core.person_spine ps
WHERE ps.occupation ILIKE ANY(ARRAY['%physician%','%doctor%','% MD%','%surgeon%'])
   OR ps.employer_sic IN ('8011','8049')

UNION ALL SELECT 'Nurses',
    COUNT(DISTINCT ps.person_id),
    COUNT(DISTINCT CASE WHEN ps.total_contributed > 0 THEN ps.person_id END),
    ROUND(SUM(ps.total_contributed))
FROM core.person_spine ps
WHERE ps.occupation ILIKE ANY(ARRAY['%nurse%',' RN%','% LPN%','%NP%'])

UNION ALL SELECT 'Veterans',
    COUNT(DISTINCT ps.person_id),
    COUNT(DISTINCT CASE WHEN ps.total_contributed > 0 THEN ps.person_id END),
    ROUND(SUM(ps.total_contributed))
FROM core.person_spine ps
WHERE ps.military_service IS NOT NULL
   OR ps.employer ILIKE ANY(ARRAY['%veteran%','%army%','%navy%','%marine%','%air force%','%VA %'])

ORDER BY total_raised DESC;
```
